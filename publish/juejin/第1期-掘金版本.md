#### SwiftUI 优势

```markdown
## 为什么选择 SwiftUI？

1. **代码量少**
   - 同样的界面，UIKit 需要 200 行，SwiftUI 只要 50 行
   
2. **开发效率高**
   - 实时预览，所见即所得
   - 声明式语法，逻辑清晰

3. **易于维护**
   - 代码结构清晰
   - 状态管理简单

对比示例:

// UIKit - 100 行
class ViewController: UIViewController {
    let label = UILabel()
    let button = UIButton()
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        setupConstraints()
        setupActions()
    }
    
    func setupUI() { ... }
    func setupConstraints() { ... }
    func setupActions() { ... }
}

struct ContentView: View {
    @State private var count = 0
    
    var body: some View {
        VStack {
            Text("Count: \(count)")
            Button("Increment") {
                count += 1
            }
        }
    }
}
```

## 实战案例

### 案例 1: 午间心灵驿站

**核心功能**: 名言推送 + 展示

**关键代码**:

```swift
// 推送管理器
class PushManager: NSObject, ObservableObject {
    func scheduleDailyNotification() {
        let content = UNMutableNotificationContent()
        content.title = "午间心灵驿站"
        content.body = "今天的至理名言已送达"
        content.sound = .default
        
        var components = DateComponents()
        components.hour = 12
        components.minute = 10
        
        let trigger = UNCalendarNotificationTrigger(
            dateMatching: components,
            repeats: true
        )
        
        let request = UNNotificationRequest(
            identifier: "daily_quote",
            content: content,
            trigger: trigger
        )
        
        UNUserNotificationCenter.current().add(request)
    }
}
```

### 案例 2: TipCalc Pro

**核心功能**: 小费计算 + AA 分摊

**关键代码**:

```swift
// 计算器管理器
class CalculatorManager: ObservableObject {
    @Published var billAmount: Double = 0.0
    @Published var tipPercent: Double = 18.0
    @Published var numberOfPeople: Int = 1
    
    var tipAmount: Double {
        billAmount * (tipPercent / 100.0)
    }
    
    var totalAmount: Double {
        billAmount + tipAmount
    }
    
    var perPersonAmount: Double {
        totalAmount / Double(numberOfPeople)
    }
}
```

```swift
// 喝水管理器
class WaterManager: ObservableObject {
    @Published var todayEntries: [WaterEntry] = []
    @Published var dailyGoal: Int = 2000
    
    var todayTotal: Int {
        todayEntries.reduce(0) { $0 + $1.amount }
    }
    
    var progressPercent: Double {
        min(Double(todayTotal) / Double(dailyGoal), 1.0)
    }
    
    func addEntry(amount: Int) {
        let entry = WaterEntry(amount: amount)
        todayEntries.append(entry)
        saveEntries()
    }
```
